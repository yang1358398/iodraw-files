```mermaid
% 清除命令窗口和工作区
clc;
clear;
close all;

% 创建图形窗口
figure('Position', [100, 100, 800, 600]);

% 定义节点坐标
nodes = [
    0, 0;    % 节点1
    -1, -1;  % 节点2
    1, -1;   % 节点3
    2, -2;   % 节点7
    3, -3;   % 节点4
    4, -4;   % 节点8
    5, -5    % 节点5
];

% 定义连接关系
edges = [
    1, 2;  % 1-2
    1, 3;  % 1-3
    3, 4;  % 3-7 (在坐标中节点7是第4个点)
    4, 5;  % 7-4 (在坐标中节点4是第5个点)
    5, 6;  % 4-8 (在坐标中节点8是第6个点)
    6, 7   % 8-5 (在坐标中节点5是第7个点)
];

% 绘制边
hold on;
for i = 1:size(edges, 1)
    plot([nodes(edges(i,1),1), nodes(edges(i,2),1)], ...
         [nodes(edges(i,1),2), nodes(edges(i,2),2)], ...
         'k-', 'LineWidth', 2);
end

% 绘制节点
scatter(nodes(:,1), nodes(:,2), 200, 'filled', ...
        'MarkerFaceColor', [0.7, 0.9, 1], ...
        'MarkerEdgeColor', 'k', ...
        'LineWidth', 2);

% 添加节点标签
node_labels = {'1', '2', '3', '7', '4', '8', '5'};
for i = 1:size(nodes, 1)
    text(nodes(i,1), nodes(i,2), node_labels{i}, ...
         'HorizontalAlignment', 'center', ...
         'VerticalAlignment', 'middle', ...
         'FontSize', 14, 'FontWeight', 'bold');
end

% 设置图形属性
title('空间图', 'FontSize', 16, 'FontWeight', 'bold');
axis equal;
grid off;
set(gca, 'XTick', []);
set(gca, 'YTick', []);
xlim([-2, 6]);
ylim([-6, 1]);

% 美化图形
set(gca, 'Box', 'on');
set(gcf, 'Color', 'white');

hold off;

```
