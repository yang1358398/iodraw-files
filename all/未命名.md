```mermaid
% 清除命令窗口和工作区
clc;
clear;
close all;

% 创建图对象
G = graph([1 1 3 7 4 8], [2 3 7 4 8 5]);

% 定义节点位置
x = [0, -1, 1, 2, 3, 4, 5];
y = [0, -1, -1, -2, -3, -4, -5];

% 绘制图形
figure('Position', [100, 100, 800, 600]);
p = plot(G, 'XData', x, 'YData', y);

% 自定义图形外观
p.MarkerSize = 8;
p.NodeColor = [0.7, 0.9, 1];
p.Marker = 'o';
p.LineWidth = 2;
p.EdgeColor = 'k';
p.NodeLabelColor = 'k';
p.NodeFontSize = 14;
p.NodeFontWeight = 'bold';

% 设置标题和坐标轴
title('空间图', 'FontSize', 16, 'FontWeight', 'bold');
axis equal;
grid off;

% 美化
set(gcf, 'Color', 'white');
```
